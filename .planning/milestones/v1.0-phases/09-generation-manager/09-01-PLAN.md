    ```typescript
    import { nanoid } from 'nanoid';
    import type { Generation, SimulationParameters, CitizenConfig } from '../schemas/index.js';
    import { GenerationSchema } from '../schemas/index.js';
    import { assignRoles } from '../roles/role-assignment.js';
    import { birthCitizen } from '../mortality/citizen-lifecycle.js';
    import { runTurns } from '../interaction/turn-runner.js';
    import { buildPeakTransmissionPrompt } from '../transmission/peak-prompt.js';
    import { executePeakTransmission } from '../transmission/transmission-executor.js';
    import { writeTransmission } from '../transmission/transmission-writer.js';
    import { mutateTransmission } from '../mutation/mutation-pipeline.js';
    import { lineageBus } from '../events/index.js';

    export async function runGeneration(
      generationNumber: number,
      params: SimulationParameters,
      seedLayer: string | null,
      recentLayer: string | null,
    ): Promise<Generation> {
      // INIT
      const generation = GenerationSchema.parse({
        id: nanoid(),
        number: generationNumber,
        phase: 'INIT',
        citizenIds: [],
        transmissionIds: [],
        startedAt: new Date().toISOString(),
      });

      // BIRTHING
      generation.phase = 'BIRTHING';
      const roles = assignRoles(params.generationSize, params.roleDistribution);
      const citizens: CitizenConfig[] = roles.map(role =>
        birthCitizen(role, generationNumber, params),
      );
      generation.citizenIds = citizens.map(c => c.id);
      lineageBus.emit('generation:started', generationNumber, citizens.length);

      // INTERACTING
      generation.phase = 'INTERACTING';
      const enrichedSeedProblem = [seedLayer, recentLayer, params.seedProblem]
        .filter(Boolean)
        .join('\n\n');
      await runTurns({ seedProblem: enrichedSeedProblem, citizens });

      // DYING -- each citizen produces a peak transmission
      generation.phase = 'DYING';
      for (const citizen of citizens) {
        const peakPrompt = buildPeakTransmissionPrompt(citizen, 0.45);
        const { transmission } = await executePeakTransmission(citizen, peakPrompt);
        await writeTransmission(transmission, params.outputDir);
        generation.transmissionIds.push(transmission.id);
        lineageBus.emit('citizen:died', citizen.id, citizen.deathProfile, citizen.generationNumber);
      }

      // TRANSMITTING -- mutate transmissions for next generation
      generation.phase = 'TRANSMITTING';
      // Re-read transmissions from the generation to mutate them is unnecessary --
      // we already have the transmission objects from the DYING phase.
      // However, the mutation pipeline creates new Transmission objects with new IDs,
      // so we need to write the mutated versions. The mutated transmission replaces
      // the original for inheritance purposes since mutateTransmission already handles
      // the no-mutation case (returns original unchanged).
      // For simplicity in v1: we don't re-write mutated transmissions to disk here.
      // The inheritance composer reads ALL files in gen directory, and we only wrote
      // the originals. Mutation is applied conceptually -- the mutateTransmission
      // call emits the transmission:mutated event and we track the result.
      // NOTE: We DO need to write mutated transmissions to disk so the inheritance
      // composer finds them. Write mutated versions alongside originals.

      // COMPLETE
      generation.phase = 'COMPLETE';
      generation.endedAt = new Date().toISOString();
      lineageBus.emit('generation:ended', generationNumber);

      return generation;
    }
    ```

    IMPORTANT implementation detail for TRANSMITTING phase: After each citizen's peak transmission is written in DYING, we DO NOT read them back from disk. Instead, we collect the Transmission objects during DYING and mutate them in TRANSMITTING. For each original transmission:
    1. Call `mutateTransmission(original, params.mutationRate, params.largeMutationProbability)`
    2. If `result.wasMutated === true`, write the mutated transmission to disk via `writeTransmission(result.transmission, params.outputDir)` -- this creates a SECOND file alongside the original
    3. The inheritance composer will read all files in the generation directory

    REVISED approach (simpler, avoids duplicate files): Write ONLY the final version. During DYING, collect transmission objects but do NOT write them yet. During TRANSMITTING, mutate each transmission, then write the final (possibly mutated) version. This ensures exactly one file per citizen.

    So the actual flow is:
    - DYING: `executePeakTransmission()` for each citizen, collect results, emit `citizen:died`
    - TRANSMITTING: `mutateTransmission()` for each collected transmission, then `writeTransmission()` for the final version

    ```typescript
    import type { SimulationParameters, Generation } from '../schemas/index.js';
    import { composeInheritance } from '../inheritance/inheritance-composer.js';
    import { lineageBus } from '../events/index.js';
    import { runGeneration } from './generation-runner.js';

    export async function runSimulation(
      params: SimulationParameters,
    ): Promise<Generation[]> {
      lineageBus.emit('simulation:started', params.seedProblem, params);
      const generations: Generation[] = [];

      for (let gen = 1; gen <= params.maxGenerations; gen++) {
        const inheritance = await composeInheritance(
          gen,
          params.outputDir,
          params.inheritanceStagingRates,
        );

        const generation = await runGeneration(
          gen,
          params,
          inheritance.seedLayer,
          inheritance.recentLayer,
        );
        generations.push(generation);
      }

      lineageBus.emit('simulation:ended', generations.length);
      return generations;
    }
    ```
